\begin{frame}
  \frametitle{Bowling Game Kata}
  \begin{itemize}
    \item Coding Kata zur täglichen Übung
      \begin{itemize}
        \item http://www.butunclebob.com/ArticleS.UncleBob.TheBowlingGameKata
      \end{itemize}
    \item Ziel: Testgetriebene Entwicklung einer Punktezähl-Logik fürs Bowling
    \item Lernziel: Perfekte Ausübung der einzelnen notwendigen Schritte zur Lösung des Problems (\textbf{Nicht} die Lösung des Problems!)
    \item Übungsform: regelmäßige Wiederholung
      \begin{itemize}
        \item Variante: Ausführung im Takt zu Musik
      \end{itemize}
    \item "`Kata"' Analogie zur Kampfkunst
  \end{itemize}
\end{frame}
\begin{frame}
  \frametitle{Bowling-Exkurs}
  \begin{itemize}
    \item Ein Spiel besteht aus 10 Frames.
    \item Jeder Frame besteht aus 2 Würfen.
    \item Pro Frame werden die gefallenen Pins als Punkte gutgeschrieben.
    \item Bei einem Spare (10 Pins mit zwei Würfen in einem Frame) wird der \emph{nächste Wurf} doppelt gezählt.
    \item Bei einem Strike (10 Pins mit einem Wurf - das beendet den
      Frame) werden die \emph{nächsten zwei Würfe} doppelt gezählt.
    \item Die Regeln für den letzten Frame sind irrelevant für die Kata (Erweiterungen sind willkomen).
  \end{itemize}
\end{frame}
\begin{frame}
  \frametitle{Tests für die Kata}
  \begin{itemize}
    \item "`Gutter Game"' (20 Würfe mit jeweils 0 Pins)
    \item "`All Ones"' (20 Würfe mit jeweils einem Pin)
    \item "`One Spare"' (Ein Spare, ein Wurf mit weniger als 10 Pins und sonst 0-Runden)
    \item "`One Strike"' (Ein Strike zwei Würfe mit zusammen weniger als 10 Pins und sonst 0-Runden)
  \end{itemize}
\end{frame}

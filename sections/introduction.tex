\begin{frame}
  \frametitle{Ziele der Versionierung}
  \begin{itemize}
    \item Protokollierung von Veränderungen
      \begin{itemize}
        \item Inklusive Metadaten wie Autor, Uhrzeit
        \item<2->\textbf{und Kommentar!}
      \end{itemize}
    \item<3->Wiederherstellung alter Zustände einzelner oder aller Dateien
    \item<4->Markierung von Zuständen als Meilensteine oder Releases
    \item<5->Koordination gemeinsamer Arbeit
    \item<6->Parallele Entwicklungszweige für Features und verschiedene Releases
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Vokabular}
  \vspace{-7mm}
  \begin{description}
    \item[SCM] Source Control Management
    \item[Arbeitskopie]<2->Alle Dateien in einer bestimmten Version
    \item[Repository]<3->Enthält alle Versionen, Entwicklungszweige, etc.
    \item[Commit]<4->Version des Projekts oder einer Datei (auch Revision)
    \item[commit]<4->Erstellen eines neuen Commits im Repository
    \item[Branch]<5->Entwicklungszweig. Mindestens \texttt{master}-Branch
    \item[Tag]<6->Markierung eines Commits, z.B. als bestimmtes Release
    \item[Konflikt]<7->Kollidierende Änderungen
    \item[Merge]<8->Zusammenführung der Änderungen mehrerer Commits
    \item[Graph]<9->Baum, der sich aus der Commit-Historie ergibt
    \item[Remote]<10->Server im Netz mit zentralem Repository
    \item[Staging]<11->Status von zum commit markierten Änderungen
  \end{description}
\end{frame}

\begin{frame}
  \frametitle{Commits}
  \begin{enumerate}
    \item Synchronisation mit dem zentralen Repository
    \item<2->Arbeit am Projekt
    \item<3->Commit der Änderungen\footnote{\tiny Bei zentralen Systemen wie SVN entfällt der Schritt an dieser Stelle und ersetzt dafür den letzten}
    \item<4->Wiederholung des ersten Schritts
    \item<5->Lösen von Konflikten
    \item<6->Upload der Commits ins zentrale Repository
  \end{enumerate}
\end{frame}
%
\begin{frame}
  \frametitle{Commits - Best Practices}
  \begin{itemize}
    \item Commit $\neq$ Undo
    \item<2->Gemeinsamer Commit zusammengehöriger Änderungen
    \item<3->Getrennter Commit verschiedener Änderungen
    \item<4->\textbf{Kein Commit von unsauberem Code}
      \begin{itemize}
        \item<5->Nicht kompilierend
        \item<5->Fehlschlagende Tests
        \item<5->Debugging-Ausgaben
      \end{itemize}
  \end{itemize}
\end{frame}
%
\begin{frame}
  \frametitle{Vorteile für die Entwickler}
  \begin{itemize}
    \item Einfache Organisation der gemeinsamen Arbeit
    \item Bei Präsentationen stets verfügbare stabile Versionen
    \item Alte Ideen konserviert
    \item Neue Ideen einfach experimentierbar
      \begin{itemize}
        \item<2->Versionskontrolle nur halbe Miete
        \item<2->Automatisierte Tests ebenfalls erforderlich (Vorlesung nächste Woche)
      \end{itemize}
  \end{itemize}
\end{frame}
